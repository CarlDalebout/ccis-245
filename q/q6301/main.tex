class Robot
{

    Robot
        : x_(0), y_(0)
    {}

    {
        ++x_;
    }

    int x_, y_;
};

int main()
{
    Robot r2d2;                               // r2d2.x_ is 0
    r2d2.walk_east();                         // r2d2.x_ is 1
    r2d2.walk_east().walk_east();             // r2d2.x_ is 3
